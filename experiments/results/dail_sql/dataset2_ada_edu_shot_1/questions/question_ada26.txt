/* Answer the following: 东三省面积最大的省是哪个，面积是多少？东三省指辽宁省、吉林省、黑龙江省。 */
Select province, Sum(Area(Shape, 1)) from cities where province in ('辽宁省', '吉林省', '黑龙江省') group by province order by Sum(Area(Shape, 1)) desc limit 1
